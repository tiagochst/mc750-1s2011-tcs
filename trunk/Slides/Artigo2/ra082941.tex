\begin{frame}{}

  % primeira transformação
  % o fim da estabilidade
  %  a partir do momento em que os computadores invadem a vida das
  %  pessoas, participando tanto dos nossos espaços públicos quanto
  %  privados perde-se a noção de localidade fixa: mobilidade
  %  Perguntas surgem: o que deve ser uma interface, o que deve
  %  permitir o usuário fazer e onde deve estar 
  %  what an interface might be, where it ism what allows a user to do
  
  \pause \begin{block}{The end of interface stability}
    \begin{itemize}
      \pause  \item    Os computadores são cada vez mais parte do nosso ambiente 
      \pause  \item    Local da interação humano-computador é desconhecido
    \end{itemize}
  \end{block}
  
  % segunda transformação
  % A segunda transformação diz a respeito da depencia criada em
  % relação aos computadores.(Declaração de imposto de renda)
  % Assim como as pessoas passaram a confiar mais nas pessoas, os
  % computadores passaram a confiar em si (nuvem - sistemas distribuidos)
  % Nuvem, internet
  % concepção de computadores: grande gama de tecnologia, sempre ligada/funcionando, sempre conectado em uma teia de sistemas  
  % tal dependencia cria novas preocupações, novas topicos de
  % pesquisa, novas oportunidades de design
  % concepção de computadores: grande gama de tecnologia, sempre ligada/funcionando, sempre conectado em uma teia de sistemas  
\vspace{-0.7cm}
  \begin{columns}[t]  

    \column{.50\textwidth}
  
  \pause \begin{block}{The growth of techno-dependency}
    \begin{itemize}
      \pause  \item  Dependência computacional
      \pause  \item  Sistemas distribuídos
    \end{itemize}
  \end{block}

  % Quarta transformação
  % O que guardar , porque guardar o que fazer com esses dados está
  % mudando paralelamente aos sistemas

  % entendiveis de uma maneira que não sabe ou vemos. 
  % copyright on digital footprint
  % Passa de um usuário desconhecido para uma pessoa, um perfil!
    \column{.50\textwidth}
      
    \uncover<11->{  \begin{block}{The end of the ephemeral}

       \uncover<12->{  \item \Gentsroom  \MVRightarrow \Info}
        \uncover<13->{ \item ``Digital footprint''  }
        \end{itemize}
      \end{block}
  }
\end{columns}
