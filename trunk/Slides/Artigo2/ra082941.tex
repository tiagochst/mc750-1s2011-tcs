\subsection{As cinco transformações}

\begin{frame}{}

  % primeira transformação
  % o fim da estabilidade da interface - The end of interface stability
  % a partir do momento em que os computadores invadem a vida das
  % pessoas, participando tanto dos nossos espaços públicos quanto
  % privados perde-se a noção de localidade fixa: mobilidade
  % Perguntas surgem: o que deve ser uma interface, o que deve
  % permitir o usuário fazer e onde deve estar 
  % what an interface might be, where it ism what allows a user to do
  
  \pause \begin{block}{O fim da estabilidade da interface}
    \begin{itemize}
      \pause  \item    Os computadores são cada vez mais parte do nosso ambiente 
      \pause  \item    Local da interação humano-computador é desconhecido
    \end{itemize}
  \end{block}
  
  % segunda transformação -The growth of techno-dependency
  % A segunda transformação diz a respeito da depencia criada em
  % relação aos computadores.(Declaração de imposto de renda)
  % Assim como as pessoas passaram a confiar mais nas pessoas, os
  % computadores passaram a confiar em si (nuvem - sistemas distribuidos)
  % Nuvem, internet
  % concepção de computadores: grande gama de tecnologia, sempre ligada/funcionando, sempre conectado em uma teia de sistemas  
  % tal dependencia cria novas preocupações, novas topicos de
  % pesquisa, novas oportunidades de design
  % concepção de computadores: grande gama de tecnologia, sempre ligada/funcionando, sempre conectado em uma teia de sistemas  
  \vspace{-0.7cm}
  \begin{columns}[t]  
    
    \column{.50\textwidth}
    \pause \begin{block}{O crescimento da dependência tecnológica}
      \begin{itemize}
        \pause  \item  Dependência computacional
        \pause  \item  Sistemas distribuídos
      \end{itemize}
    \end{block}
    
    % Quarta transformação -The end of the ephemeral
    % O que guardar , porque guardar o que fazer com esses dados está
    % mudando paralelamente aos sistemas
    % Dados pessoais -> Digital figenprint -> passamos a ser visiveis e
    % entendiveis de uma maneira que não sabe ou vemos. 
    % copyright on digital footprint
    % Passa de um usuário desconhecido para uma pessoa, um perfil!
    \column{.50\textwidth}
    
    \uncover<11->{  \begin{block}{O fim do efêmero}
        \begin{itemize}
          \uncover<12->{  \item \Gentsroom  \MVRightarrow \Info}
          \uncover<13->{ \item ``Digital footprint''  }
        \end{itemize}
      \end{block}
    }
  \end{columns}
  

  \vspace{-0.2cm} 
  \begin{columns}[t] 
    % terceira transformação - The growth in hyperconnectivity
    \column{.50\textwidth}
    \uncover<8-> {
      \begin{block}{O cresciemento da hiperconectividade}
        \begin{itemize}
          % \item Habilidade das novas ferramentas de melhorar a eficiência e   ganhar tempo
          % \item<alert@1> Mas, há um aumento no consumo de tempo devido a presença digital
          \uncover<9->{ \item Novas ferramentas \MVRightarrow \hspace{0.5mm}  Eficiência e ganho tempo}
          \uncover<10->{   \item<alert@1>  Presença digital  \MVRightarrow \hspace{0.5mm} Consumo de tempo  }
          
        \end{itemize}
      \end{block}
    }

    % quinta transformação - The growth of creative engagement
    % Novas tipos de ferramentas - pode-se trabalhar, jogar e expressarmos
    % de novas maneira
    % Antigamento computadores limita-se em mecanização e
    % automatização de tarefas relacionadas trabalhos
    % Instrumentos para criatividade -> programadores, podutores, editores
    \column{.50\textwidth} 
    \uncover<14->{ 
      \begin{block}{O crescimento da participação criativa}
        \begin{itemize}
          \uncover<15->{\item  Mecanização e automatização de tarefas  }
          \uncover<16->{\item  Novas ferramentas \MVRightarrow
            \hspace{0.5mm}  Criatividade}
        \end{itemize}
      \end{block}
    }
  \end{columns}
  
\end{frame}
