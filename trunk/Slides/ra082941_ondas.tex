\begin{frame}
  \frametitle{Paradigmas}
  \begin{itemize}
    \pause \item O que é um paradigma?
    \begin{itemize}
    \item <1-> Maneira descrever ondas de pesquisa em determinado campo
    \item <1-> Novos paradigmas não constestam paradigmas anteriores
    \item <1-> Paradigmas co-existem
    \end{itemize}

    \pause \item Quais elementos de um paradigma em IHC?
    \begin{itemize}
    \item <1-> Um entendimento comum das notáveis propriedades da
      interação
   \item <1-> Tipos de questões interessantes  sobre as propriedades
    \item <1-> Procedimentos para responder as perguntas
    \item <1-> Conhecimento comum de como interpretar os resultados
      dos procedimentos

    \end{itemize}
  \end{itemize}
\end{frame}
